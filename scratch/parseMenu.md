const categories = [
  "Soups", "Salads", "Sides", "Chatar Patar (veg)", "Chatar Patar (non-veg)",
  "Starters", "Noodles", "Dim Sum", "Chinese Mains & Thai", "International Delicacies",
  "Pasta", "Pizza", "From Korea", "Raita", "Across The Globe",
  "Indian Main Course (veg)", "Indian Main Course (non-veg)", "Breads", "Desserts",
  "Rice And Biryani", "House Specials"
];

const parseMenu = (text) => {
  const lines = text.split(\'\\n\').map(l => l.trim()).filter(l => l !== \'\');
  const items = [];
  let currentCategory = "";

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];

    if (categories.includes(line)) {
      currentCategory = line;
      continue;
    }

    if (line.includes(\'₹\')) {
      const name = lines[i - 1];
      const price = parseFloat(line.replace(\'₹\', \'\').replace(\',\', \'\'));
      let description = "";
      let isVeg = false;

      let j = i + 1;
      while (j < lines.length && !categories.includes(lines[j]) && !lines[j].includes(\'₹\') && !(j + 1 < lines.length && lines[j + 1].includes(\'₹\'))) {
        if (lines[j] === "Vegetarian") {
          isVeg = true;
        } else {
          description = (description + " " + lines[j]).trim();
        }
        j++;
      }

      if (name && !categories.includes(name)) {
        items.push({
          id: Math.random().toString(36).substr(2, 9),
          name,
          price,
          description,
          category: currentCategory,
          is_veg: isVeg,
          is_spicy: false
        });
      }
      i = j - 1;
    }
  }
  return items;
};

const result = parseMenu(menuText);
